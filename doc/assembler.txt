Assembler Syntax
================
; This is a simple comment
; All symbol is limited to 63 character long

A==0		; Define A as hexdecimal 0
.ADDR	100	; Starting Address (Hex)
START:	CLR
	LOAD	%A, TEMP
	HLT
COUNT:	CLR
	INC	%A
	J	R,Z,+2	; Relative condition jump (signed integer)
	J	R,Z,-2
	JMP START
TEMP:	#0	; Note that constants are in hexdecimal, not octal
	.ASCII	"Hello, World!\n"	; ASCII Encoded Text
