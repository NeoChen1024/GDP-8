; Assembler Syntax
; ================
; This is a simple comment
; All symbol is limited to 63 character long

.DEF	A	R00		; Define
.DEF	B	R01
.ADDR	$100			; Starting Address (Hex)
START:	CLR
	LOAD	W, %A, TEMP	; Normal
	LOAD	W, %A, TEMP:%B	; Indexed
	LI	%A, #0		; Intermediate
	HLT
COUNT:	CLR			; Clear all flag
	INC	%A		; Increment, if result is 0, set Z flag
	IO	CPU, DPY, %A	; Display data
	JR	Z, +2		; Relative condition jump (signed integer)
	JR	X, -2		; Unconditionally relative jump
	J	@NOPE		; @ means indirect access
	JR	X, @NOPE	; Same effect as above

TEMP:	.WORD	$0		; Note that constants are in hexdecimal, not octal
HW:	.ASCII	"WOW\n"		; ASCII Encoded Text
ABYTE:	.BYTE	$FF		; One byte
NOPE:	.ADDR	START		; Contains 3-byte address
